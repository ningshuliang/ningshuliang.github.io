    测试CSI
    import numpy as np
import cv2
import os
import sys
import numpy as np


def evaluation_CSI(path):
    SCI20 = []
    SCI30 = []
    SCI40 = []
    SCI50 = []
    SCI = [SCI20, SCI30, SCI40, SCI50]
    boundary = [20, 30, 40, 50]
    for k in range(len(boundary)):
        border = boundary[k]
        print(border)
        seqs = os.listdir(path)
        for j in range(10):
            a = b = c = d = 0
            for seq in seqs:
                first = cv2.imread(path + '/' + seq + '/' + 'target' + str(j).zfill(2) + '.png', 0)
                first = np.array(first)
                mask1 = first > 80
                first = 0 * mask1 + (1 - mask1) * first

                second = cv2.imread(path + '/' + seq + '/' + 'test' + str(j).zfill(2) + '.png', 0)
                second = np.array(second)
                mask2 = second > 80
                second = second * (1 - mask2) + 0 * mask2

                np_a = (first >= border + 0) * (second >= border + 0)
                a += np_a.sum()
                np_b = (first < border + 0) * (second >= border + 0)
                b += np_b.sum()
                np_c = (first >= border + 0) * (second < border + 0)
                c += np_c.sum()
                np_d = (first < border + 0) * (second < border + 0)
                d += np_d.sum()
            a = a/len(seqs)
            b = b/len(seqs)
            c = c/len(seqs)
            d = d/len(seqs)
            score = a / (a + b + c + 0.000001)
            print(score)
            SCI[k].append(score)
    return SCI
    光流法插帧
    import cv2,time,pickle
import os
import numpy as np

def warp_flow(img, flow):
    h, w = flow.shape[:2]
    flow = -flow
    flow[:,:,0] += np.arange(w)
    flow[:,:,1] += np.arange(h)[:,np.newaxis]
    res = cv2.remap(img, flow, None, cv2.INTER_LINEAR)
    return res
