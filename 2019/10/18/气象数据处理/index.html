def separate_train_valid(dir_path, valid_dir_path, train_proportion=0.95):
    """
    Assumes directory structure: dir_path has folders named TRAIN_#, each TRAIN_# contains folders
    of image sequences.
    This function randomly selects train_proportion of them to remain in original set, while move the
    others to VALID_# under valid_dir_path
    """
    if not os.path.exists(valid_dir_path):
        os.mkdir(valid_dir_path)
    train_folders = os.listdir(dir_path)
    valid_folders = [folder_name.replace('TRAIN', 'VALID') for folder_name in train_folders]
    train_folders = [os.path.join(dir_path, folder_name) for folder_name in train_folders]
    valid_folders = [os.path.join(valid_dir_path, folder_name) for folder_name in valid_folders]
    for train_folder, valid_folder in zip(train_folders, valid_folders):
        if not os.path.exists(valid_folder):
            os.mkdir(valid_folder)
        seq_names = os.listdir(train_folder)
        for seq_name in seq_names:
            if random.random() > train_proportion:
                shutil.move(os.path.join(train_folder, seq_name), os.path.join(valid_folder, seq_name))

    im2rec_path = mx.test_utils.get_im2rec_path()

        subprocess.check_call(['python', im2rec_path, '--list', '--recursive', '--test-ratio=0.2', prefix_path, data_path],
                            stdout=devnull)

    # sort the lst file
    if num_parts == 1:
        with open(prefix_path+'.lst', 'r+') as f:
            f.seek(0)
            lines = f.readlines()
            lines.sort(key=lambda line:re.split(r'\s+', line)[-2])
            f.seek(0)
            f.writelines(lines)
        # generate record file
        with open(os.devnull, 'wb') as devnull:
            subprocess.check_call(['python', im2rec_path, '--num-thread=4', '--pass-through', '--test-ratio=0.2', prefix_path, data_path],
                            stdout=devnull)
