    import numpy as np
import os
import shutil
import sys

files=os.listdir('First')
newfile=sorted(files, key=lambda x: x[:12])
filelist=[]
total=len(newfile)
print(total)
index=0
while index < total:
    print(index)
    if index==total-1:
        filelist.append(newfile[index])
        index+=1
    else:
        first=newfile[index][:12]
        second=newfile[index+1][:12]
        if first==second:
            index11=newfile[index].find('2500_')
            index12=newfile[index].find('.ref')
            index21=newfile[index+1].find('2500_')
            index22=newfile[index+1].find('.ref')
            number1=int(newfile[index][index11+5:index12])
            number2=int(newfile[index+1][index21+5:index22])
            if number1<number2:
                filelist.append(newfile[index+1])
            else:
                filelist.append(newfile[index])
            index+=2
        else:
            filelist.append(newfile[index])
            index+=1
if not os.path.exists('Second'):
    os.makedirs('Second')
for i in filelist:
    os.rename('First/'+i,'First/'+i[:12]+'.ref')
    try:
        shutil.move('First/'+i[:12]+'.ref',"Second/")
    except:
        print('Here')
print('Done!')
